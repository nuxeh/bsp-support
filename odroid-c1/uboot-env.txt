kernel_addr_r=0x21000000
fdt_addr_r=0x21800000
kernel_file=uImage
dtb_file=meson8b-odroidc1.dtb
load_kernel=fatload mmc 0:1 $(kernel_addr_r) $(kernel_file)
load_dtb=fatload mmc 0:1 $(fdt_addr_r) $(dtb_file)
bootargs=console=ttyAML0,115200 root=/dev/mmcblk0p2 rootwait
do_boot=bootm $(kernel_addr_r) - $(fdt_addr_r)
bootcmd=run load_kernel load_dtb do_boot
